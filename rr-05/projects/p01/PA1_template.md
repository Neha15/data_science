## What is mean total number of steps taken per day?

```r
calcStepsPerDay <- function(data) {
    stepsPerDay <- ddply(
        data, 
        .(date), 
        function(x) {  
            sum(x$steps, na.rm = T)
        }
    )
    
    return(stepsPerDay)
}

stepsPerDay <- calcStepsPerDay(new_data)

plot(x = stepsPerDay$date, y = stepsPerDay$V1, type= "h")
```
